#include <iostream>
#include <vector>
#include <iomanip>
#include <cmath>
using namespace std;

void print_state(const vector<bool>& is_prime, int n, int p) {
    cout << "After processing p = " << p << ":\n";
    for (int i = 2; i <= n; i++) {
        if (is_prime[i]) cout << setw(3) << i;   // keep primes
        else cout << setw(3) << ".";             // crossed out
    }
    cout << "\n\n";
}

void sieve(int n) {
    vector<bool> is_prime(n+1, true);
    is_prime[0] = is_prime[1] = false;

    for (int p = 2; p * p <= n; p++) {
        if (is_prime[p]) {
            // mark multiples of p
            for (int k = p * p; k <= n; k += p) {
                is_prime[k] = false;
            }
            print_state(is_prime, n, p);
        }
    }

    cout << "Final primes up to " << n << ":\n";
    for (int i = 2; i <= n; i++) {
        if (is_prime[i]) cout << i << " ";
    }
    cout << "\n";
}

int main() {
    int n = 30;
    sieve(n);
    return 0;
}
